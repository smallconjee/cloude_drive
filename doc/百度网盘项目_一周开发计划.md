## 2 二、数据库设计：四表分离

```sql
CREATE DATABASE IF NOT EXISTS cloud_dirve DEFAULT CHARSET=utf8mb4;
USE cloud_dirve;

-- ==========================================
-- 1. 用户表 (users)
-- ==========================================
CREATE TABLE IF NOT EXISTS users (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    username    VARCHAR(64)  NOT NULL UNIQUE,
    password    VARCHAR(128) NOT NULL,             -- crypt() SHA-512 加密后
    salt        VARCHAR(32)  NOT NULL,             -- 盐值（每用户不同）
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- 2. 真实文件元数据 (real_files) → 类似 inode
-- ==========================================
CREATE TABLE IF NOT EXISTS real_files (
    file_md5    VARCHAR(64) PRIMARY KEY,           -- 整个文件的MD5作为主键
    file_size   BIGINT NOT NULL,                   -- 文件总大小
    chunk_count INT NOT NULL,                      -- 分了多少个块
    ref_count   INT DEFAULT 0,                     -- 引用计数（秒传 & 安全删除）
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- 3. 块数据 (chunks)
-- ==========================================
CREATE TABLE IF NOT EXISTS chunks (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    file_md5    VARCHAR(64) NOT NULL,              -- 属于哪个文件
    chunk_index INT NOT NULL,                      -- 块序号（0, 1, 2...）
    chunk_md5   VARCHAR(64) NOT NULL,              -- 单块MD5（用于校验）
    chunk_size  INT NOT NULL,                      -- 块大小（最后一块可能不足1MB）
    real_path   VARCHAR(512) NOT NULL,             -- 物理存储路径
    FOREIGN KEY (file_md5) REFERENCES real_files(file_md5),
    UNIQUE KEY  idx_file_chunk (file_md5, chunk_index)
);

-- ==========================================
-- 4. 虚拟文件/目录项 (virtual_files) → 类似 dentry
-- 优化：parent_dir_md5 CHAR(32) 做索引，避免 VARCHAR(512) 索引溢出
-- ==========================================
CREATE TABLE IF NOT EXISTS virtual_files (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    user_id         INT NOT NULL,
    parent_dir      VARCHAR(512) NOT NULL DEFAULT '/',  -- 原文保留，给人看
    parent_dir_md5  CHAR(32) NOT NULL,                  -- MD5 值，给索引用
    filename        VARCHAR(255) NOT NULL,
    is_dir          TINYINT DEFAULT 0,                   -- 0=文件，1=目录
    file_md5        VARCHAR(64),                         -- 指向 real_files（目录为空）
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    UNIQUE KEY  idx_path (user_id, parent_dir_md5, filename)
);
CREATE INDEX idx_ls ON virtual_files(user_id, parent_dir_md5);
```

### 2.1 核心业务流向

```
秒传：  客户端算MD5 → 查 real_files → 命中 → virtual_files 加记录 + ref_count++

上传：  CMD_UPLOAD_INIT → 切块循环发 CMD_UPLOAD_CHUNK → CMD_UPLOAD_DONE
        服务端每收一块：存磁盘 + 写 chunks 表
        全部收完：写 real_files + virtual_files

断点续传：重连后 → 服务端查 chunks 表已有的块序号 → 客户端跳过这些块

下载：  客户端开 N 个线程 → 每个线程建独立连接 → 请求不同 chunk
        → pwrite(fd, buf, size, offset) 并发写入本地文件

删除：  删 virtual_files → ref_count-- → 若 ref_count==0 → 删 chunks + 删物理文件
```

## 3 三、通信协议设计

### 3.1 包结构

```
┌────────────────────────────────────────────────────┐
│  Magic      (4B)   0x434C4452 ("CLDR")             │  ← 识别合法包
├────────────────────────────────────────────────────┤
│  Version    (1B)   0x01                            │  ← 协议版本
├────────────────────────────────────────────────────┤
│  cmd        (4B)   命令码 / 状态码                   │
├────────────────────────────────────────────────────┤
│  data_len   (4B)   变长数据区长度                    │
├────────────────────────────────────────────────────┤
│  file_md5   (65B)  文件MD5（传输时用，其他命令留空）    │
├────────────────────────────────────────────────────┤
│  chunk_idx  (4B)   块序号（分块传输时用）              │
├────────────────────────────────────────────────────┤
│  token      (33B)  会话Token（登录后的请求必带）       │
├────────────────────────────────────────────────────┤
│  data       (变长) 数据区                            │
├────────────────────────────────────────────────────┤
│  CRC32      (4B)   对 Magic~data 的 CRC32 校验       │  ← 传输层完整性
└────────────────────────────────────────────────────┘
```

### 3.2 C 结构体定义

```c
// ---- protocol.h ----

#define PROTO_MAGIC   0x434C4452  // "CLDR"
#define PROTO_VERSION 0x01
#define CHUNK_SIZE    (1 * 1024 * 1024)  // 1MB

// 命令码
enum CMD_TYPE {
    // 用户
    CMD_REGISTER      = 1,
    CMD_LOGIN         = 2,
    CMD_LOGOUT        = 3,
    CMD_HEARTBEAT     = 4,
    
    // 目录操作（短命令 → 快池）
    CMD_MKDIR         = 10,
    CMD_LS            = 11,
    CMD_CD            = 12,
    CMD_PWD           = 13,
    CMD_RM            = 14,
    
    // 文件传输（长命令 → 慢池）
    CMD_UPLOAD_INIT   = 20,
    CMD_UPLOAD_CHUNK  = 21,
    CMD_UPLOAD_DONE   = 22,
    CMD_DOWNLOAD_INIT = 30,
    CMD_DOWNLOAD_CHUNK= 31,
};

// 状态码
enum RESP_CODE {
    RESP_OK           = 0,
    RESP_FAIL         = 1,
    RESP_FAST_UPLOAD  = 2,   // 秒传成功
    RESP_RESUME       = 3,   // 断点续传（数据区带已存在的块序号列表）
    RESP_AUTH_FAIL    = 4,
    RESP_NOT_FOUND    = 5,
};

// 包头（定长部分）
typedef struct {
    uint32_t magic;         // 0x434C4452
    uint8_t  version;       // 0x01
    int      cmd;           // 命令码 或 状态码
    int      data_len;      // 变长数据区长度
    char     file_md5[65];  // 文件MD5
    int      chunk_index;   // 块序号
    char     token[33];     // 会话Token
} packet_header_t;

// 发送流程：
// 1. send(fd, &header, sizeof(header))
// 2. send(fd, data, header.data_len)      // 变长数据
// 3. send(fd, &crc32, sizeof(uint32_t))   // CRC32 校验
```

### 3.3 双线程池命令分发

```c
// main.c 中 epoll 事件循环收到新连接后：
int connfd = accept(sockfd, NULL, NULL);

// 先 recv 一个 packet_header_t
packet_header_t header;
recv(connfd, &header, sizeof(header), MSG_WAITALL);

// 校验 Magic
if (header.magic != PROTO_MAGIC) { close(connfd); continue; }

// 根据命令码分发到不同线程池
if (header.cmd >= 20) {
    // 长命令（上传/下载）→ 慢池
    enQueue(&slow_pool.que, connfd);
    pthread_cond_signal(&slow_pool.cond);
} else {
    // 短命令（ls/cd/mkdir/login...）→ 快池
    enQueue(&fast_pool.que, connfd);
    pthread_cond_signal(&fast_pool.cond);
}
```

## 4 四、核心模块详细设计

### 4.1 时间轮（Timing Wheel）心跳检测

```
原理：一个环形数组（比如60格，每格1秒），每个格子挂一条链表。
    
     指针每秒走一格
         ↓
    ┌─┬─┬─┬─┬─┬─┬─┬─┐
    │0│1│2│3│...   │59│
    └─┴─┴─┴─┴─┴─┴─┴─┘
     │       │
    [连接A] [连接B, 连接C]    ← 每格挂着「在这一刻该超时」的连接
    
收到心跳：把连接从旧格子摘下来，挂到「当前指针 + 超时秒数」的格子上
指针走到某格：把这格上所有连接全部断开（说明超时没收到心跳）
```

```c
// timing_wheel.h
#define TW_SLOTS   60     // 60个槽位，对应60秒
#define TW_TIMEOUT 30     // 超时30秒

typedef struct tw_node {
    int fd;
    struct tw_node *prev, *next;  // 双向链表
} tw_node_t;

typedef struct {
    tw_node_t *slots[TW_SLOTS];   // 环形数组，每格一个链表头
    int cursor;                    // 当前指针位置
    pthread_mutex_t lock;
} timing_wheel_t;

void tw_init(timing_wheel_t *tw);
void tw_add(timing_wheel_t *tw, int fd);       // 新连接加入
void tw_refresh(timing_wheel_t *tw, int fd);   // 收到心跳，刷新位置
void tw_tick(timing_wheel_t *tw);              // 指针前进一格，清理超时连接
```

### 4.3 客户端多线程并行下载

```c
// download.h
typedef struct {
    char     server_ip[64];
    char     server_port[16];
    char     file_md5[65];
    int      output_fd;         // 本地文件 fd（多线程共享）
    int     *chunk_list;        // 本线程负责的块序号数组
    int      chunk_count;       // 负责几个块
    char     token[33];         // 会话 Token
} download_task_t;

void *download_thread(void *arg) {
    download_task_t *task = (download_task_t *)arg;
    
    // 每个线程独立建连接
    int sockfd = connect_to_server(task->server_ip, task->server_port);
    
    for (int i = 0; i < task->chunk_count; i++) {
        int idx = task->chunk_list[i];
        
        // 请求第 idx 块
        send_download_chunk_req(sockfd, task->file_md5, idx, task->token);
        
        // 接收块数据
        char buf[CHUNK_SIZE];
        int received = recv_chunk_data(sockfd, buf);
        
        // pwrite：写到精确偏移，无需加锁
        off_t offset = (off_t)idx * CHUNK_SIZE;
        pwrite(task->output_fd, buf, received, offset);
    }
    
    close(sockfd);
    return NULL;
}

// 主线程：
//   1. 发 CMD_DOWNLOAD_INIT 获取 chunk_count
//   2. 创建 N 个线程，每个分配不同的 chunk 序号
//   3. pthread_join 等全部完成
//   4. md5sum 校验本地文件
```

### 4.4 Token 会话管理

```c
// session.h — 服务端维护的会话表
typedef struct session_node {
    char token[33];           // 32字节随机 hex 字符串
    int  user_id;
    char username[64];
    char cur_dir[512];        // 用户当前所在虚拟目录
    time_t last_active;
    struct session_node *next;
} session_t;

// 用哈希表存储（token → session_t）
// 登录成功时：生成随机 token → 插入哈希表 → 返回给客户端
// 后续请求：从 packet_header_t.token 取出 → 查哈希表 → 得到 user_id
// 登出 / 超时：从哈希表删除
```

### 4.5 守护进程化

```c
// 在现有 main.c 的 fork 之后，子进程中加入：
void daemonize() {
    // 1. 创建新会话，脱离终端
    setsid();
    
    // 2. 再次 fork，确保不会重新获取终端
    if (fork() != 0) exit(0);
    
    // 3. 关闭标准输入/输出/错误
    close(STDIN_FILENO);
    close(STDOUT_FILENO);
    close(STDERR_FILENO);
    
    // 4. 重定向到 /dev/null（防止后续代码写 stdout 出错）
    open("/dev/null", O_RDONLY);  // fd 0
    open("/dev/null", O_WRONLY);  // fd 1
    open("/dev/null", O_WRONLY);  // fd 2
    
    // 5. 所有输出改走日志系统（彩色终端输出在 daemon 模式下自动关闭）
}
```
